public class Resta {
	
	private double restar1(double num1, double num2) { 
		double resta = 0;
		resta = num1 - num2;
		return resta;
	}
	
	
	private int restar2(int num1, int num2) { 
		int resta = 0;
		resta = num1 - num2;
		return resta;
	}

	private double restar3(double num1, double num2,double num3) { 
		double resta = 0;
		resta = num1 - num2 - num3;
		return resta;
	}
	
	private double restar4(double num1, int veces) { 
		double acumulado = 0;
		
		for (int i = 0; i < veces; i++) {
			acumulado = acumulado + i;
			
		}
		return acumulado;
		
	}
}
